class Solution {
public:
    int rearrangeCharacters(string s, string target) {
        int ans = 101;
        vector<int> f(26);
        vector<int> f2(26);
        for (auto& c: s) {
            f[c - 'a']++;
        }
        for (auto& c: target) {
            f2[c - 'a']++;
        }
        for (int i = 0; i < 26; i++) {
            if (f2[i] > 0) 
            ans = min(ans, f[i] / f2[i]);
        }
        return ans;
    }
};

class Solution {
public:
    string doubleToString(const double &dbNum) {
        char *chCode;
        chCode = new(std::nothrow)char[20];
        sprintf(chCode, "%.2lf", dbNum);  // .2 是控制输出精度的，两位小数
        string strCode(chCode);
        delete []chCode;
        return strCode;
    }
    string discountPrices(string s, int di) {
        string ans = "";
        vector<string> word;
        string t;
        for(auto c:s){
            if(c == ' '){
                word.push_back(t);
                t.clear();
            } else {
                t+=c;
            }
        }
        word.push_back(t);
        vector<string> f;
        for (auto& w: word) {
            if (w[0] != '$') {
                f.push_back(w);
            } else {
                bool flag = true;
                for (int i = 1; i < w.length(); i++) {
                    if (w[i] == '$' || (w[i] >= 'a' && w[i] <= 'z')) {
                        flag = false;
                        break;
                    }
                }
                if (w.length() == 1) flag = false;
                if (!flag) {
                    f.push_back(w);
                } else {
                    string t = w.substr(1, w.length() - 1);
                    double tt = 0;
                    long long car = 1;
                    for (int i = t.length() - 1; i >= 0; i--) {
                        tt += (t[i] - '0') * car;
                        car *= 10;
                    }
                    double d = 1.0 * tt * (100 - di) / 100.00;
                    string ttt = "$";
                    ttt += doubleToString(d);
                    f.push_back(ttt);
                }
            }
        }
        for (int i = 0; i < f.size(); i++) {
            if (i > 0) ans += " ";
            ans += f[i];
        }
        return ans;
    }
};
